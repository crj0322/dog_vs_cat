def get_cam(img, img_size, MODEL, model_name, input_shape, preprocess_func):
    # base model
    base_model = MODEL(weights='imagenet', include_top=False)
    
    # top layers
    top_model = load_model('top_model_%s.h5'%model_name)
    activation_weights = top_model.layers[-1].get_weights()
    
    # predict
    img = load_img(img, target_size=input_shape)
    img = img_to_array(img)
    x = np.expand_dims(img, axis=0)
    x = preprocess_func(x)
    feature_map = base_model.predict(x)
    feature_map = np.squeeze(feature_map)
    pred = np.average(feature_map, axis=(0, 1))
    pred = np.matmul(pred, activation_weights[0]) + activation_weights[1]
    pred = 1 / (1 +  np.e ** (-pred))
    
    # class activateion map
    cam = (pred - 0.5) * np.matmul(feature_map, activation_weights[0])
    cam = (cam - cam.min()) / (cam.max() - cam.min())
    cam = cv2.resize(cam, img_size)
    cam = 255 * cam
    cam = cam.astype(np.uint8)
    
    return pred, cam

def mix_img(img, cam, threshold=100):
    img = img_to_array(img).astype(np.uint8)
    heatmap = cv2.applyColorMap(cam, cv2.COLORMAP_JET)
    heatmap[np.where(cam <= threshold)] = 0

    # overlapping: alpha*src1 + beta * src2 + gamma
    out = cv2.addWeighted(src1=img, alpha=0.8, src2=heatmap, beta=0.2, gamma=0)
    return out

def plot_mix(pred, mix):
    if pred < 0.5:
        text = 'cat %.2f%%' % (100 - pred * 100)
    else:
        text = 'dog %.2f%%' % (pred * 100)

    text = model_name + ' ' + text
    plt.imshow(mix)
    plt.title(text)
